If needed:
sudo docker build -t localization_ros2 .

Terminal 1:
sudo docker run -it --rm \
  --name localization_container \
  --network host \
  --ipc=host \
  --pid=host \
  --privileged \
  -e RMW_IMPLEMENTATION=rmw_cyclonedds_cpp \
  -e ROS_DOMAIN_ID=0 \
  -e ROS_LOCALHOST_ONLY=0 \
  -e CYCLONEDDS_URI=file:///cyclonedds.xml \
  -v /home/ubuntu/ros2_ws/cyclonedds.xml:/cyclonedds.xml \
  -v /home/ubuntu/map:/map \
  -v /home/ubuntu/last_path_bag:/last_path_bag \
  localization_ros2

(Check: 'ros2 topic hz /velodyne_points' has to have an average rate of ~10)

ros2 launch lidar_localization_ros2 lidar_localization.launch.py

Terminal 2:
sudo docker exec -it localization_container bash

option 1) path input (saving last /path msg to last_path_bag):
    ros2 run jeep_driver last_path_recorder_node

option 2) autonomous path following (using /path msg from last_path_bag):
    ros2 run jeep_driver path_follower_node
    
Debugging:
ros2 topic echo /pcl_pose
ros2 topic echo /path_drive
